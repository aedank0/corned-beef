136 static constexpr std::size_t ONE_LESS_SIZE = sizeof(std::size_t) - 1;
137 static constexpr std::size_t LOG2_SIZE = 2 | (sizeof(std::size_t) >> 3);
138 static constexpr std::size_t SIZE_BITS = sizeof(std::size_t) >> LOG2_SIZE;
139 static constexpr std::size_t SIZE_BITS_HALF = sizeof(std::size_t) >> 1;
140 static constexpr std::size_t SIZE_BITS_SUB_ONE = SIZE_BITS - 1;
141 static constexpr std::size_t SIZE_BITS_SUB_ONE_LSH_TWO = SIZE_BITS_SUB_ONE << 2;
142
143 //END CONSTANT ZONE

151 static constexpr std::size_t NextNDivBySize(std::size_t val)
152 {
153 return (val + ONE_LESS_SIZE) & ~ONE_LESS_SIZE;
154 }

164 static constexpr std::size_t CombineHashes(std::size_t a, std::size_t b, int rotAmt = static_cast<int>(SIZE_BITS_HALF))
165 {
166 return a ^ std::rotl(b, rotAmt);
167 }

176 template<std::same_as<std::size_t>... Args>
177 static constexpr std::size_t CombineHashes(Args... args)
178 {
179 auto impl = [&]<std::size_t... I>(std::index_sequence<I...>) constexpr
180 {
181 std::size_t retVal = (std::rotl(args, I * std::max(SIZE_BITS / sizeof...(Args), std::size_t(1))) ^ ...);
182
183 return retVal;
184 };
185
186 return impl(std::index_sequence_for<Args...>());
187 }

219 template<TrivialCopy T>
220 struct Hash<T>
221 {
228 constexpr std::size_t operator()(const T& val) const
229 {
230 const std::size_t size = NextNDivBySize(sizeof(T));
231 const std::size_t count = size >> LOG2_SIZE;
232 std::size_t data[count] = {0};
233
234 std::memcpy(data, &val, sizeof(T));
235
236 std::size_t retVal = data[0];
237
238 for (size_t i = 1; i < count; ++i)
239 {
240 retVal = CombineHashes(retVal, data[i], i & SIZE_BITS);
241 }
242
243 return CB_XOR_VALUE(retVal);
244 }
245 };
