133 static constexpr std::size_t ONE_LESS_SIZE = sizeof(std::size_t) - 1;
134 static constexpr std::size_t LOG2_SIZE = 2 | (sizeof(std::size_t) >> 3);
135 static constexpr std::size_t SIZE_BITS = sizeof(std::size_t) >> LOG2_SIZE;
136 static constexpr std::size_t SIZE_BITS_SUB_ONE = SIZE_BITS - 1;
137 static constexpr std::size_t SIZE_BITS_SUB_ONE_LSH_TWO = SIZE_BITS_SUB_ONE << 2;
138
139 //END CONSTANT ZONE
140
147 static constexpr std::size_t NextNDivBySize(std::size_t val)
148 {
149 return (val + ONE_LESS_SIZE) & ~ONE_LESS_SIZE;
150 }
151
160 static constexpr std::size_t CombineHashes(std::size_t a, std::size_t b, int rotAmt = 32)
161 {
162 return a ^ std::rotl(b, rotAmt);
163 }

170 template<TrivialCopy T>
171 struct Hash<T>
172 {
179 constexpr std::size_t operator()(const T& val) const
180 {
181 const std::size_t size = NextNDivBySize(sizeof(T));
182 const std::size_t count = size >> LOG2_SIZE;
183 std::size_t data[count] = {0};
184
185 std::memcpy(data, &val, sizeof(T));
186
187 std::size_t retVal = data[0];
188
189 for (size_t i = 1; i < count; ++i)
190 {
191 retVal = CombineHashes(retVal, data[i], i & SIZE_BITS);
192 }
193
194 return CB_XOR_VALUE(retVal);
195 }
196 };
